type WorkerNode struct {
	Endpoint string
	Semaphore chan struct{}
	//The semaphore: a channel with a buffer size = max concurrent requests eg 1. 
}

var modelRouteMap = make(map[string][]*WorkerNode)
var routeMutex sync.RWMutex

func refreshTopology() {
	
	log.Println("scanning network topology...")
	
	//1. Read JSON file (Same as before) 
	file, err := os.ReadFile("topology.json")
	if err != nil {
		log.Printf("Could not read topology.json: %v", err)
		setupFallbackLocalhost()
		return
	}
		
	var topo TopologyConfig
	
	if err := json.Unmarshal(file, &topo); err != nil {
		log.Printf("unmarshalling fail")
		setupFallbackLocalhost() 
		return 
	}
	
	//2. prepare new maps. 
	newModelMap := make(map[string][]*WorkerNode)
		
	getWorker := func(endpoint string, priority int) *WorkerNode {
		routeMutex.Lock()
		defer routeMutex.Unlock()
		
		if existing, ok := allNodes[endpoint]; ok {
			return existing 
		}
		
		capacity := 1 //if you trust a node to handle 2, set to 2.
		
		newNode := &WorkerNode {
			Endpoint: endpoint, 
			Semaphore: make(chan struct{}, capacity),
		}
		allNodes[endpoint] = newNode
		return newNode 
	}
	
		//Helper to scan a node
	scanNode := func(endpoint string) {
		resp, err := http.Get(endpoint + "/api/tags")
		if err != nil { return }
		defer resp.Body.Close()
		
		var tags OllamaTagsResponse
		if err := json.NewDecoder(resp.Body).Decode(&tags); err != nil { return } 
		
		nodeRef :=  getWorker(endpoint)
		
		for _, m := range tags.Models {
			//append this node to models's list of providers
		newModelMap[m.Name] = append(newModelMap[m.Name], nodeRef)
		log.Printf("topology: model '%s' available at %s", m.Name, endpoint)
		}
	}
	
	//3. execute scans
	scanNode(topo.Local.Endpoint)
	for _, remote := topo.Remotes {
		scanNode(remote.Endpoint)
	}
	
	//4. update routing table
	routeMutex.Lock()
	modelRouteMap = newModeMap
	routeMutex.Unlock()
		
	
}

func acquireNode(modelName string) (string, func()){
	//replaces pickBestNode
//returns a node endpoint and a release function. 
//this function BLOCKS if no nodes are available. 
//1. get candidates
routeMutex.RLock()
candidates := modelRouteMap[modelName]
routeMutex.RUnlock()

if len(candidates) == 0 {
return "", nil
}
//2. the select loop
//we randomly shuffly to be fair but we grab a token from any candidate. 
for {
	for _, node := range candidates {
		select {
			case node.Semaphore <- struct{}{}:
			//SUCCESS pushed token into the channel. Node is locked
			log.Printf("acquired %s for %s", node.Endpoint, modelName)
			
			//return the endpoint and the release closure 
			return node.Endpoint, func() {
				//to release we pop the token out
				<- node.Semaphore
				log.Printf("released %s", node.Endpoint)
			}
			default: 
//channel is full (node is busy. continue to next candidate	
			continue
		}
	}
//3. The wait state 
//if we looped all candidates and they're all busy. 
//wait before trying again to avoid CPU spinning. 
//condition variable. 
time.Sleep(60000 * time.Millisecond)
}

}

func handleSmartChatProxy(w http.ResponseWriter, r *http.Request){
	if r.Method != http.MethodPost {
		http.Error(w, "Only POST allowed", http.StatusMethodNotAllowed)
		return
	}
	
	//1. peek body 
	bodyBytes, _ := io.ReadAll(r.Body)
	defer r.Body.Close()
	
	type ModelPeek struct {Model string `json:"model"`}
	var peek ModelPeek
	json.Unmarshal(bodyBytes, &peek)
	
	//2. SERVICE ROUTING (virtualLLM)
	if strings.HasPrefix(peek.Model, "gemini") { 
	//acquire google semaphor. 
	googleSemaphore <- struct{}{}
	defer func() { <- googleSemaphore }()
	r.Body = io.NopCloser(bytes.NewBuffer(bodyBytes))
	handleGoogleAIProxy(w,r)
	return 
	}
	//3. Ollama routing //this line blocks until node is free. The browser connection stays open, spinning its wheel. 
	endpoint, releaseFunc := acquireNode(peek.Model)
	if endpoint == "" {
	//fallback to localhost default logic if not specific route found. assume localhost is in topology.json anyway. 
	endpoint = "http://localhost:11434"
	} else { 
	defer releaseFunc() 
	}
	
	//4. forwarding the API caller 
	targetURL := endpoint + ollamaAPIPath
	proxyReq, _ := http.NewRequestWithContext(r.Context(), http.MethodPost, targetURL, bytes.NewBuffer(bodyBytes))
	proxyReq.Header.Set("Content-Type", r.Header.Get("Content-Type"))
	
	client := http.DefaultClient
	resp, err := client.Do(proxyReq)
	if err != nil {
		http.Error(w, "Node unreachable", 502)
		return 
	}
	defer resp.Body.Close()
	
	copyHeader(w.Header(), resp.Header)
	w.WriteHeader(resp.StatusCode)
	flusher, _ := w.(http.Flusher)
	io.Copy(flushWriter{w, flusher}, resp.Body)
}

//copyHeaders copies specific headers from source to destination/skips certain headers like connection which are hopbyhop
func copyHeader(dst, scr http.Header){
	for k, vv := range scr {
		//filter hopbyhop headers? )often handled by server/client// EG: if k == "Connection" || k == "Proxy-Connection" 
		for _, vb := range vv {
			dst.Add(k,vb)
		}
	}
}


// --- Proxy Handler Logic ---
// handleOllamaProxy forwards requests to the Ollama server and strewams the response.
func handleOllamaProxy(w http.ResponseWriter, r *http.Request){
	//1. validate request method (POST)
	if r.Method != http.MethodPost {
		http.Error(w, "Method Not Allowed: Only POST requests are accepted", http.StatusMethodNotAllowed)
		log.Printf("Received nonPOST request on /api/ollama from %s",r.RemoteAddr)
	return
	}
	//2. Prepare the request to Ollama
	ollamaURL := ollamaServerURL + ollamaAPIPath

//read the body from the incoming request (from UX.html)
//read it into a buffer first in case its needed later or if underlying stream needs closing
clientRequestBody, err := io.ReadAll(r.Body)
if err != nil {
	http.Error(w, "failed to read request body", http.StatusInternalServerError)
	log.Printf("Error reading request body from %s: %v", r.RemoteAddr, err)
return
}
defer r.Body.Close()// close the original request

//create a new request to the ollama server
proxyReq, err := http.NewRequestWithContext(r.Context(), http.MethodPost, ollamaURL, bytes.NewReader(clientRequestBody))
if err != nil {
	http.Error(w, "failed to create proxy request", http.StatusInternalServerError)
	log.Printf("error creating proxy request %v",err)
	return
}	

//3. copy Headers
//copy necessary headers from the client request to proxy request
proxyReq.Header.Set("Content-Type", r.Header.Get("Content-Type"))
proxyReq.Header.Set("Accept", r.Header.Get("Accept"))//often "application/json" for streaming. 
//set the host header correctly for ollama server
targetURL, _ := url.Parse(ollamaServerURL)
proxyReq.Host = targetURL.Host 
//anything else? 

log.Printf("proxying request for %s to %s", r.RemoteAddr, ollamaURL)

//4. Execute the request to Ollama
//Use a client with potentially adjusted timeouts if needed; DefaultClient respects context.
client := http.DefaultClient
ollamaResp, err := client.Do(proxyReq)
if err != nil {
	//handle some errors
	if os.IsTimeout(err) {
		http.Error(w,"gateway timeout: ollama server timed out", http.StatusGatewayTimeout)
		log.Printf("Timeout connecting to Ollama %v", err)
	}else{
		http.Error(w,"bad gateway: error contacting ollama server", http.StatusBadGateway)
		log.Printf("error connecting to ollama %v", err)
	}
return
}
defer ollamaResp.Body.Close() 

log.Printf("recived response from ollama %d", ollamaResp.StatusCode)

//5. Stream the response back to the original client (browser)
//Copy ollama response headers
copyHeader(w.Header(), ollamaResp.Header)
//write the ollama status code to ur response writer
w.WriteHeader(ollamaResp.StatusCode)

flusher, ok := w.(http.Flusher)
if !ok {
	log.Println("Warning: ResponseWriter doesnt support flushing")
}

//use io.copy to efficiently stream the body//io.copy handles chucked encoding automatically if present in ollamaResp
bytesCopied, err := io.Copy(flushWriter{w,flusher}, ollamaResp.Body)
if err != nil {
	//this error might occur if the client disconeccted
	log.Printf("Error streaming ollama sresponse to client %s:%v",r.RemoteAddr, err)
}
log.Printf("finished streaming response to %s (%d)", r.RemoteAddr, bytesCopied)
}

//flushWriter is a helper to flush the response writer after each write call. This ensures chunks are sent immediately over the network. 
type flushWriter struct {
	w http.ResponseWriter
	flusher http.Flusher
}

func (fw flushWriter) Write (p []byte) (n int, err error){
	n, err = fw.w.Write(p)//write the data
	if fw.flusher != nil {
		fw.flusher.Flush()//flush data to the client
	}
	return 
}

func getActiveOpenAIKey() string {
	keyMutex.Lock()
	defer keyMutex.Unlock()
	if storedOpenAIKey == "" {
		return ""
	}
	if time.Now().After(storedKeyExpiry){
		storedOpenAIKey = ""
		storedKeyExpiry = time.Time{}
		return ""
	}
	return storedOpenAIKey
}
	


func handleOpenAIProxy(w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		http.Error(w, "Only POST allowed", http.StatusMethodNotAllowed)
		return
	}
	
	apiKey := getActiveOpenAIKey();
	if apiKey == "" {
		http.Error(w, "No valid OpenAI API key stored", http.StatusUnauthorized)
	return
	}
	//read incoming JSON body from browser. 
	body, err := io.ReadAll(r.Body)
	if err != nil {
		http.Error(w, "Failed to read body", 500)
		return
	}
	
	defer r.Body.Close()
	
	//Forward request to ChatGPT API
	req, err := http.NewRequest( 
	"POST", 
	"https://api.openai.com/v1/chat/completions", 
	bytes.NewReader(body),
	)
	if err != nil {
		http.Error(w, "Failed to create request", 500)
	return
	}

	req.Header.Set("Content-Type", "application/json")
	req.Header.Set("Authorization", "Bearer "+apiKey)
	
	client := http.Client{}
	resp, err := client.Do(req)
	if err != nil {
		http.Error(w, "Error contacting OpenAI", 502)
		return
	}
	defer resp.Body.Close()
	
	//Prepare to stream SSE back to browser
	w.Header().Set("Content-Type", "text/event-stream")
	w.Header().Set("Cache-Control", "no-cache")
	w.Header().Set("Connection", "keep-alive")
	
	flusher, ok := w.(http.Flusher)
	if !ok {
		http.Error(w, "Streaming unsupported", 500)
		return
	}
	
	reader := bufio.NewReader(resp.Body)
	for {
		line, err := reader.ReadString('\n')
		if err != nil {
			break
		}
		w.Write([]byte(line))
		flusher.Flush()
	}
}

	apiKey := strings.TrimSpace(string(body))
	if apiKey == "" {
		http.Error(w, "Empty API Key", http.StatusBadRequest)
		return
	}
	
	keyMutex.Lock()
	storedOpenAIKey = apiKey
	storedKeyExpiry = time.Now().Add(keyTimeout)
	keyMutex.Unlock()
	
	w.WriteHeader(http.StatusOK)
	w.Write([]byte("Key stored"))
}

//Helper struct to read the incoming Quipu/Ollama JSON format
type OllamaRequest struct {
	Model string `json: "model"`
	Messages []struct {
		Role string `json:"role"`
		Content string `json:"content"`
	} `json:"messages"`
	Stream bool `json:"stream"`
}

//Helper struct to write outgoing JSON chunks (mimicking Ollama response)
type OllamaResponse struct {
	Model string `json:"model"`
	Message struct {
		Role string `json:"role"`
		Content string `json:"content"`
	} `json:"messages"`
	Done bool `json:"done"`
}

func handleGoogleAIProxy (w http.ResponseWriter, r *http.Request) {
	if r.Method != http.MethodPost {
		http.Error(w, "Method Not Allowed", http.StatusMethodNotAllowed)
		return
	}
	
	//1. Validate Key Existence and Expiry
	keyMutex.Lock()
	apiKey := storedGoogleAIKey
	keyMutex.Unlock()
	
	if apiKey == "" {
		http.Error(w, "API Key not found. Please authenticate.", http.StatusUnauthorized)
		return
	}
	
	/*
	expiry := storedKeyExpiry
	if time.Now().After(expiry){
		//Auto-clear expired key
		keyMutex.Lock()
		storedGoogleAIKey = ""
		keyMutex.Unlock()
		http.Error(w, "API Key expired. Please  re-authenticate", http.StatusUnauthorized)
		return
	} */

//2. parse incoming request from quipu

	var req OllamaRequest 
 if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
//	var req GoogleProxyRequest
//	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		http.Error(w, "invalid JSON body", http.StatusBadRequest)
		return
	}
	//default to flash if model is missing
	if req.Model == "" {
		req.Model = "gemini-2.0-flash"
	}
	
	//3. Initialize GenAI Client using the stored key 
	ctx := r.Context()
client, err := genai.NewClient(ctx, &genai.ClientConfig{APIKey: apiKey, Backend: genai.BackendGeminiAPI, })
if err != nil {
	log.Printf("Failed to create GenAI client: %v", err)
	http.Error(w, "Failed to initialize AI client", http.StatusInternalServerError)
	return
}

var promptBuilder strings.Builder
for _, msg := range req.Messages {
	promptBuilder.WriteString(fmt.Sprintf("Role: %s ; Content: %s ;", msg.Role, msg.Content))
}




flusher, ok := w.(http.Flusher)
if !ok {
	http.Error(w, "Streaming not supported", http.StatusInternalServerError)
return
}

//5. send response back to quipu//we wrap the raw text in a JSON object for easier parsing in JS
//responseObj := GoogleProxyResponse{ Content: result.Text(),}

w.Header().Set("Content-Type", "application/json")
w.Header().Set("Transfer-Encoding", "chunked")

//4. Generate Content 
log.Printf("Proxying request to google AI [model: %s]", req.Model)
stream4 := client.Models.GenerateContentStream(
ctx, 
req.Model,
genai.Text(promptBuilder.String()),
nil,
)
 

//json.NewEncoder(w).Encode(responseObj)
for resp, err := range stream4 {
	if err != nil {
		sendErrorChunk(w, flusher, req.Model, err)
	return
	 }
	if err != nil {
		log.Printf("Google API Error: %v", err)
		//Send error as content so user sees it in UI
		errJSON, _ := json.Marshal(OllamaResponse{
Model: req.Model, 
Message: struct { Role string "json:\"role\""; Content string "json:\"content\"" }{
		Role: "assistant", Content: fmt.Sprintf("\n[Error: %v]",err)},
Done: true,
		})
w.Write(errJSON)
w.Write([]byte("\n"))
flusher.Flush()
return
	}


var chunkText string
if len(resp.Candidates) > 0 {
	for _, part := range resp.Candidates[0].Content.Parts {
	chunkText += part.Text
	}
}
if chunkText != "" {
	sendResponseChunk(w, flusher, req.Model, chunkText, false)
}
}
sendResponseChunk(w, flusher, req.Model, "", true)
 }
 
 
 func sendResponseChunk(w http.ResponseWriter, f http.Flusher, model, text string, done bool) {
	 resp :=  OllamaResponse{
Model: model,
Done: done,
	 }

if !done {
resp.Message.Role = "assistant"
resp.Message.Content = text
}	

jsonBytes, _ := json.Marshal(resp)
w.Write(jsonBytes)
w.Write([]byte("\n"))
f.Flush()
 
 }
 
 func sendErrorChunk(w http.ResponseWriter, f http.Flusher, model string, err error){
	 resp := OllamaResponse {
		 Model: model, 
		 Done: true,
	 }
	 resp.Message.Role = "assistant"
	 resp.Message.Content = fmt.Sprintf("\n[Error: %v]",err)
	 
	 jsonBytes, _ := json.Marshal(resp)
	 w.Write(jsonBytes)
	 w.Write([]byte("\n"))
	 f.Flush()
 }
